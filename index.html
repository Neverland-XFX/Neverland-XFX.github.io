public int LengthOfLongestSubstring(string s) {
    int maxLen = 0;
    for (int i = 0; i < s.Length; i++) {
        var set = new HashSet<char>();
        int j = i;
        while (j < s.Length && !set.Contains(s[j])) {
            set.Add(s[j]);
            j++;
        }
        maxLen = Math.Max(maxLen, j - i);
    }
    return maxLen;
}
